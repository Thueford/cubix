\lstdefinestyle{style_cs}{
    language=cs, style=codecolors, morekeywords={
        attribute, xsl, if
    }
}

\newcommand{\csinline}[1]{\lstinline[style=style_cs]{#1}}

\newcommand{\csmlinline}[1]{
    \lstinline[style=style_cs,style=codeblock]{#1}
}

\lstnewenvironment{cs}[1][]{\lstset{style=style_cs, style=codeblock1, #1}}{}
\lstnewenvironment{cs2}[1][]{\lstset{style=style_cs, style=codeblock2, #1}}{}

\newcommand*{\csitem}[1]{
  \setbox0\hbox{\csinline{#1}}
  \item[\usebox0] \hfill \\
}

\newcommand*{\csmlitem}[1]{
	\setbox0\hbox{\csmlinline{#1}}
	\item[\usebox0] \hfill \\
}
